% paper.tex — Sutra paper LaTeX wrapper.
%
% The substantive paper content lives in paper/paper.md. The pages.yml
% workflow runs `pandoc paper.md -t latex -o paper.tex.body` before
% latexmk, so the entire body of the markdown lands here at \input
% time. paper.tex itself is a thin wrapper that supplies the
% documentclass, packages, anonymization switch, title, and author
% block.
%
% Two builds from the same source via the \ifanon switch:
%   pdflatex paper.tex                                    -> named (full)
%   pdflatex -usepretex='\newif\ifanon\anontrue' paper.tex -> anonymized
%
% Build chain:
%   pages.yml  -> pandoc paper.md -t latex -o paper.tex.body
%              -> latexmk -pdf paper.tex      (named)
%              -> latexmk -pdf -usepretex=... paper.tex (anonymized)
%              -> docs/paper.pdf staged for the GitHub Pages deploy

% Built with lualatex so Unicode (−, ≈, ∞, ⁹, etc.) lands as-is.
\documentclass[11pt]{article}
\usepackage{fontspec}
% microtype is intentionally not loaded: it queries character-level
% protrusion factors for every glyph it sees, and the
% \newunicodechar block below substitutes some characters with
% multi-token math expressions (\ensuremath{\approx} etc.) that
% microtype can't resolve into a single slot. The result was a
% "! Missing number" error per substituted glyph occurrence —
% non-fatal under nonstopmode, but they accumulated and pushed
% lualatex to exit code 1 even after a clean PDF was written.
% Without microtype, lualatex exits 0 cleanly. The cost is some
% hairline kerning improvements; the paper renders fine without it.
\usepackage{hyperref}
\usepackage{url}
\usepackage{xcolor}
\usepackage{geometry}
\geometry{margin=1in}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{array}
% pandoc's longtable column-width syntax uses arithmetic on
% \linewidth ("(\linewidth - 8\tabcolsep) * \real{0.6098}") which
% needs both the calc package AND the \real macro from pandoc's
% default preamble. Without these, every multi-column table
% (capacity sweeps, per-axis layout, Appendix E sweeps, Appendix A
% L-sweep) renders the column-width fractions as literal text and
% the cells collapse into one wrapped column ("() * 0.0976 ro- ta-
% tion k=8"). With them, the tables render cleanly.
\usepackage{calc}
\providecommand{\real}[1]{#1}
\usepackage{caption}
\usepackage{enumitem}
\usepackage{verbatim}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,shapes.geometric,calc,fit,backgrounds}
% Map the math-style Unicode used in prose ($\approx$, $\times$, $\in$,
% superscript digits, etc.) onto the LaTeX equivalents lmroman has
% glyphs for. Without this, lualatex emits "missing character" warnings
% and (when combined with cross-reference resolution passes) returns
% exit code 1, which kills latexmk. lualatex DOES handle unicode
% natively, but only for glyphs the loaded font actually contains —
% lmroman has no text-mode ≈ / ⁻ / ⁵ / ∈ / ≠ / ‖ / → / − etc.
\usepackage{newunicodechar}
\newunicodechar{≈}{\ensuremath{\approx}}
\newunicodechar{≠}{\ensuremath{\neq}}
\newunicodechar{×}{\ensuremath{\times}}
\newunicodechar{−}{\ensuremath{-}}
\newunicodechar{∈}{\ensuremath{\in}}
\newunicodechar{‖}{\ensuremath{\Vert}}
\newunicodechar{→}{\ensuremath{\rightarrow}}
\newunicodechar{…}{\ensuremath{\ldots}}
\newunicodechar{‑}{-}
\newunicodechar{⁰}{\textsuperscript{0}}
\newunicodechar{¹}{\textsuperscript{1}}
\newunicodechar{²}{\textsuperscript{2}}
\newunicodechar{³}{\textsuperscript{3}}
\newunicodechar{⁴}{\textsuperscript{4}}
\newunicodechar{⁵}{\textsuperscript{5}}
\newunicodechar{⁶}{\textsuperscript{6}}
\newunicodechar{⁷}{\textsuperscript{7}}
\newunicodechar{⁸}{\textsuperscript{8}}
\newunicodechar{⁹}{\textsuperscript{9}}
\newunicodechar{⁻}{\textsuperscript{\ensuremath{-}}}
% pandoc emission shims:
\providecommand{\tightlist}{\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}
\providecommand{\pandocbounded}[1]{#1}
% pandoc's `Shaded` (background-shaded code block) — minimal definition
% so the workflow doesn't need pandoc's full default preamble. We just
% wrap code in a `quote`-styled block; full syntax highlighting is
% disabled via pandoc's --no-highlight.
\usepackage{framed}
\definecolor{shadecolor}{RGB}{248,248,248}
\newenvironment{Shaded}{\begin{snugshade*}}{\end{snugshade*}}

\hypersetup{
  colorlinks=true,
  linkcolor=blue,
  urlcolor=blue,
  citecolor=blue,
}
